<!-- ✅ 540p canvas -->

<canvas id="board" width="960" height="540"></canvas>


<script>

const canvas = document.getElementById("board");

const ctx = canvas.getContext("2d");


const size = 8;

const cellW = canvas.width  / size;

const cellH = canvas.height / size;


/* ====== STATE ====== */

let startTime = performance.now();

let phase = 0;           // 0..1 interpolation factor

const duration = 1000;   // transition duration in ms


/* random colors per cell */

let colorsA = [];

let colorsB = [];


/* ====== UTILS ====== */

function randomColor() {

  return [

    Math.random() * 255,

    Math.random() * 255,

    Math.random() * 255

  ];

}


function lerp(a, b, t) {

  return a + (b - a) * t;

}


function lerpColor(c1, c2, t) {

  return [

    lerp(c1[0], c2[0], t),

    lerp(c1[1], c2[1], t),

    lerp(c1[2], c2[2], t)

  ];

}


function initColors() {

  colorsA = [];

  colorsB = [];

  for (let i = 0; i < size * size; i++) {

    colorsA.push(randomColor());

    colorsB.push(randomColor());

  }

}


/* ====== RENDER ====== */

function drawBoard(alpha) {

  let i = 0;

  for (let y = 0; y < size; y++) {

    for (let x = 0; x < size; x++, i++) {

      const c = lerpColor(colorsA[i], colorsB[i], alpha);

      ctx.fillStyle = `rgb(${c[0] | 0}, ${c[1] | 0}, ${c[2] | 0})`;

      ctx.fillRect(

        x * cellW,

        y * cellH,

        cellW,

        cellH

      );

    }

  }

}


/* ====== ANIMATION LOOP ====== */

function animate(now) {

  const elapsed = now - startTime;

  phase = Math.min(elapsed / duration, 1);


  drawBoard(phase);


  if (phase >= 1) {

    // When transition finishes, prepare the next one

    startTime = now;

    colorsA = colorsB;

    colorsB = [];

    for (let i = 0; i < size * size; i++) {

      colorsB.push(randomColor());

    }

  }


  requestAnimationFrame(animate);

}


/* ====== START ====== */

initColors();

requestAnimationFrame(animate);


/* ====== exportFrame (for iOS) ====== */

window.exportFrame = function () {

  const imageData = ctx.getImageData(

    0, 0, canvas.width, canvas.height

  );

  // Returns raw RGBA pixels as ArrayBuffer

  return imageData.data.buffer;

};

</script>
